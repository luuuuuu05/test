using System.Collections;
using UnityEngine;
using UnityEngine.XR;

public static class HapticHelper
{
    static InputDevice? GetRightHand()
    {
        var devices = new List<InputDevice>();
        InputDevices.GetDevicesWithCharacteristics(
            InputDeviceCharacteristics.Right | InputDeviceCharacteristics.HeldInHand, devices);
        return devices.Count > 0 ? devices[0] : null;
    }

    public static void SendHaptic(float amplitude, float duration)
    {
        var device = GetRightHand();
        if (device.HasValue) device.Value.SendHapticImpulse(0u, amplitude, duration);
    }

    public static void Grab()           => SendHaptic(0.6f, 0.15f);
    public static void GrabGold()       => SendHaptic(1.0f, 0.30f);
    public static void Score()          => SendHaptic(0.2f, 0.08f);
    public static void CountdownTick()  => SendHaptic(0.1f, 0.04f);
    public static void CountdownGo()    => SendHaptic(0.8f, 0.25f);
    public static void GameWin()
    {
        var coroutineHost = GameObject.FindObjectOfType<MonoBehaviour>();
        if (coroutineHost != null) coroutineHost.StartCoroutine(WinPattern());
    }
    static IEnumerator WinPattern()
    {
        SendHaptic(0.5f, 0.1f); yield return new WaitForSeconds(0.15f);
        SendHaptic(0.7f, 0.1f); yield return new WaitForSeconds(0.15f);
        SendHaptic(1.0f, 0.2f);
    }
    public static void ButtonClick()    => SendHaptic(0.08f, 0.02f);
    public static void PlaceObject()    => SendHaptic(0.15f, 0.06f);
    public static void GrabFailed()     => SendHaptic(0.3f, 0.05f);
}

using UnityEngine;

public static class ParticleFactory
{
    public static ParticleSystem SpawnBurst(Vector3 worldPos, Color color, int count = 12, float duration = 0.8f)
    {
        var go = new GameObject("VFX_Burst");
        go.transform.position = worldPos;
        var ps = go.AddComponent<ParticleSystem>();
        var main = ps.main;
        main.duration = duration;
        main.loop = false;
        main.startLifetime = 0.6f;
        main.startSpeed = 2f;
        main.startSize = 0.15f;
        main.startColor = color;
        main.simulationSpace = ParticleSystemSimulationSpace.World;
        
        var emission = ps.emission;
        emission.rateOverTime = 0;
        emission.SetBursts(new[] { new ParticleSystem.Burst(0f, count) });
        
        var shape = ps.shape;
        shape.shapeType = ParticleSystemShapeType.Sphere;
        shape.radius = 0.3f;

        var renderer = go.GetComponent<ParticleSystemRenderer>();
        renderer.material = new Material(Shader.Find("Particles/Standard Unlit"));
        renderer.renderMode = ParticleSystemRenderMode.Billboard;

        ps.Play();
        Object.Destroy(go, duration + 0.5f);
        return ps;
    }

using UnityEngine;
using UnityEngine.XR;

public class HandAnimatorDriver : MonoBehaviour
{
    public Animator handAnimator;
    public XRNode handNode = XRNode.RightHand;

    void Update()
    {
        if (handAnimator == null) return;
        var device = InputDevices.GetDeviceAtXRNode(handNode);
        device.TryGetFeatureValue(CommonUsages.grip, out float grip);
        device.TryGetFeatureValue(CommonUsages.trigger, out float trigger);
        handAnimator.SetFloat("Grip", grip);
        handAnimator.SetFloat("Trigger", trigger);
    }
}

private int _displayScore;
private Coroutine _scoreAnim;

private IEnumerator AnimateScore(int target)
{
    int start = _displayScore;
    float t = 0f;
    while (t < 0.4f)
    {
        t += Time.deltaTime;
        _displayScore = (int)Mathf.Lerp(start, target, t / 0.4f);
        scoreText.text = $"Score: {_displayScore}";
        if (t < 0.2f) scoreText.color = Color.Lerp(Color.white, Color.yellow, t / 0.2f);
        else scoreText.color = Color.Lerp(Color.yellow, Color.white, (t - 0.2f) / 0.2f);
        yield return null;
    }
    _displayScore = target;
    scoreText.text = $"Score: {_displayScore}";
    scoreText.color = Color.white;
}

private IEnumerator AnimateCountdown(int count)
{
    string text = count > 0 ? count.ToString() : "GO!";
    countdownText.text = text;
    countdownText.color = count > 0 && count <= 5 ? Color.red : Color.white;
    
    float t = 0f;
    while (t < 0.5f)
    {
        t += Time.deltaTime;
        float s = 1f + Mathf.Sin(t / 0.5f * Mathf.PI) * 0.5f;
        countdownText.transform.localScale = Vector3.one * s;
        yield return null;
    }
    countdownText.transform.localScale = Vector3.one;
}

任务 C-7：旗子网络同步动画

public void RemoveFlag(string flagId)
{
    if (_flags.TryGetValue(flagId, out var flag))
    {
        Debug.Log($"[Flag] Removed {flagId}");
        // 播放缩小消失动画
        StartCoroutine(AnimateRemove(flag));
    }
}

IEnumerator AnimateRemove(FlagObject flag)
{
    float t = 0f;
    Vector3 origScale = flag.transform.localScale;
    while (t < 0.3f)
    {
        t += Time.deltaTime;
        flag.transform.localScale = Vector3.Lerp(origScale, Vector3.zero, t / 0.3f);
        yield return null;
    }
    _flags.Remove(flag.FlagID);
    Destroy(flag.gameObject);
}
